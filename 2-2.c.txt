#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

int main() {
	int size = 0, query = 0;
	int *arr = (int*)malloc(sizeof(int)*size);

	printf("값의 크기 입력 : ");
	scanf("%d", &size);

	printf("값을 공백으로 구분해 입력 : ");
	for(int i = 0; i < size; i++) {
		scanf("%d", &arr[i]);
	}

	printf("쿼리 크기 입력 : ");
	scanf("%d", &query);

	for (int i = 0; i < query; i++) {
		int L = 0, R = 0, sum = 0;

		scanf("%d %d", &L, &R);

		for (int j = L - 1; j < R; j++) {
			sum += *(arr + j);
		}
		
		printf("%d\n", sum);
	}

	return 0;
}
